# Get Your Pelican APK in 5 Minutes — GitHub Auto-Build

No Android Studio needed. GitHub builds it FREE automatically.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
STEP 1 — Create a FREE GitHub account
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
→ https://github.com/signup
(use any email, takes 60 seconds)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
STEP 2 — Create a new repository
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. Click the "+" icon (top right) → "New repository"
2. Name it: pelican-avvc
3. Set to PRIVATE (your project data stays private)
4. Click "Create repository"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
STEP 3 — Upload this ZIP
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. On your new repo page, click "uploading an existing file"
2. Drag ALL files from this unzipped folder into the page
   (select all → drag, or use the file picker)
3. Scroll down → click "Commit changes"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
STEP 4 — Watch the APK being built (3 min)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. Click the "Actions" tab at the top of your repo
2. You'll see "Build Pelican AV/VC APK" running
3. Wait ~3 minutes (GitHub builds it on their servers)
4. Click the completed run → scroll down → "Artifacts"
5. Download "Pelican-AVVC-Release-APK"  ← YOUR APK IS HERE

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
STEP 5 — Install on Android phones
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
• Send APK via WhatsApp or email to team members
• On Android: Settings → Security → Install unknown apps → Allow
• Tap the APK file → Install
• "Pelican AV/VC" appears in the app drawer with your logo ✅

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
UPDATING THE APP IN FUTURE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. Edit index.html on GitHub (pencil icon)
2. Commit → new APK builds automatically in 3 min
3. Download new APK, share with team
